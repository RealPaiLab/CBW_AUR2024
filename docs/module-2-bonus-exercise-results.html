Color-code tSNE plot by species, try various perplexity levels:

species_cols = c("orchid","forestgreen")[factor(crabs$sp)]
par(mfrow=c(2,2))
plot(c_tsne10[,1],
     c_tsne10[,2],
     main = "Perplexity = 10",
     col = species_cols)

plot(c_tsne20[,1],
     c_tsne20[,2],
     main = "Perplexity = 20",
     col = species_cols)

     c_tsne50[,2],
     main = "Perplexity = 50",
     col = species_cols)

     c_tsne100[,2],
     main = "Perplexity = 100",
     col = species_cols)
